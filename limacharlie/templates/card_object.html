$def with (oid, oname, otype, olocs, locs, rlocs, parents, children, sid, hostnames)

Parents: $oname ($otype)

$if sid is not None: As observed on $sid
$for pid, pname, ptype, prel in parents:
Type Name # Obj Loc # Rel Loc
$ptype $pname $locs.get( pid, 'N/A' ) $rlocs.get( pid, 'N/A' ) share arrow_drop_down_circle

Locations: $oname ($otype)

Globally on: $locs.get( oid, '-' ) hosts

$for lsid, lastseen in olocs:
Sensor Hostname Last Seen
$lsid $hostnames.get( lsid, '-' ) $msTsToTime( lastseen )

Children: $oname ($otype)

$if sid is not None: As observed on $sid
$for cid, cname, ctype, crel in children:
Type Name # Obj Loc # Rel Loc
$ctype $cname $locs.get( cid, 'N/A' ) $rlocs.get( cid, 'N/A' ) share arrow_drop_down_circle
